\section{Question}
\begin{frame}{Question}
    Show that if $AB = \cbrak{\phi}$, then $\pr{A} \leq \pr{B'}$
\end{frame}

\section{Solution}
\subsection{Data Given from Question}
\begin{frame}{Data Given from Question}
    \begin{block}{Theory}
        \begin{align}
            \label{eq:Theory_1}
            \pr{A+B} = \pr{A} + \pr{B} - \pr{AB}
        \end{align}
    \end{block}
    Given,
    \begin{align}
        AB &= 0
        \\
        \label{eq:1}
        \implies \pr{AB} &= 0
    \end{align}
\end{frame}

\subsection{Result}
\begin{frame}{Result}
    By \eqref{eq:1} and \eqref{eq:Theory_1},
    \begin{align}
        \pr{A+B} = &\pr{A} + \pr{B} \leq 1 
        \\
        \implies &\pr{A} + 1 - \pr{B'} \leq 1
        \\
        \label{eq:Result}
        \implies &\pr{A} \leq \pr{B'}
    \end{align}
    By \eqref{eq:Result}, Proved. 
\end{frame}
